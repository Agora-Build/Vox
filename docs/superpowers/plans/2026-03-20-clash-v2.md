# Clash v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rebuild Clash as an event-driven system with warm runner pool, live spectator dashboard (audio + transcript + metrics), and central moderator orchestration.

**Architecture:** Events are the top-level container (every match belongs to an event). PipeWire handles local agent-to-agent audio for measurement accuracy. Agora RTC handles spectator streaming and moderator. A WebSocket hub relays live metrics and transcript from runner to spectators. Runners live in a warm pool with long-lived tokens.

**Tech Stack:** Express + Drizzle ORM (backend), React + TanStack Query (frontend), Agora RTC + ConvoAI (streaming/moderator), PipeWire (local audio), `ws` (WebSocket hub), Playwright (browser agents), Vitest + Playwright (tests)

**Spec:** `docs/superpowers/specs/2026-03-20-clash-v2-design.md`

```typescript
import { WebSocketServer, WebSocket } from "ws";
import type { Server } from "http";

interface SpectatorConnection {
  ws: WebSocket;
  matchId: number;
}

interface RunnerConnection {
  ws: WebSocket;
  matchId: number;
}

const spectators = new Map<number, Set<SpectatorConnection>>();
const runners = new Map<number, RunnerConnection>();

export function setupClashWebSocket(httpServer: Server): void {
  const wss = new WebSocketServer({ noServer: true });

  httpServer.on("upgrade", (request, socket, head) => {
    const url = new URL(request.url || "", `http://${request.headers.host}`);

    // /ws/clash-runner/:matchId — runner metrics stream
    const runnerMatch = url.pathname.match(/^\/ws\/clash-runner\/(\d+)$/);
    if (runnerMatch) {
      wss.handleUpgrade(request, socket, head, (ws) => {
        const matchId = parseInt(runnerMatch[1]);
        runners.set(matchId, { ws, matchId });
        ws.on("message", (data) => {
          // Relay runner metrics to all spectators of this match
          const conns = spectators.get(matchId);
          if (conns) {
            const msg = data.toString();
            for (const conn of conns) {
              if (conn.ws.readyState === WebSocket.OPEN) {
                conn.ws.send(msg);
              }
            }
          }
        });
        ws.on("close", () => runners.delete(matchId));
      });
      return;
    }

    // /ws/clash/:matchId — spectator connection
    const spectatorMatch = url.pathname.match(/^\/ws\/clash\/(\d+)$/);
    if (spectatorMatch) {
      wss.handleUpgrade(request, socket, head, (ws) => {
        const matchId = parseInt(spectatorMatch[1]);
        if (!spectators.has(matchId)) {
          spectators.set(matchId, new Set());
        }
        const conn: SpectatorConnection = { ws, matchId };
        spectators.get(matchId)!.add(conn);

        // Send current spectator count
        broadcastToSpectators(matchId, {
          type: "spectatorCount",
          count: spectators.get(matchId)!.size,
        });

        ws.on("close", () => {
          spectators.get(matchId)?.delete(conn);
          if (spectators.get(matchId)?.size === 0) {
            spectators.delete(matchId);
          } else {
            broadcastToSpectators(matchId, {
              type: "spectatorCount",
              count: spectators.get(matchId)?.size || 0,
            });
          }
        });
      });
      return;
    }

    // Not a clash WebSocket — destroy
    socket.destroy();
  });
}

export function broadcastToSpectators(matchId: number, data: unknown): void {
  const conns = spectators.get(matchId);
  if (!conns) return;
  const msg = JSON.stringify(data);
  for (const conn of conns) {
    if (conn.ws.readyState === WebSocket.OPEN) {
      conn.ws.send(msg);
    }
  }
}

export function sendToRunner(matchId: number, data: unknown): void {
  const conn = runners.get(matchId);
  if (conn && conn.ws.readyState === WebSocket.OPEN) {
    conn.ws.send(JSON.stringify(data));
  }
}

export function getSpectatorCount(matchId: number): number {
  return spectators.get(matchId)?.size || 0;
}
```

      const { matchId, metricsA, metricsB, recordingUrl, durationSeconds, error: matchError } = req.body;
      if (!matchId) return res.status(400).json({ error: "matchId required" });

      const match = await storage.getClashMatch(matchId);
      if (!match) return res.status(404).json({ error: "Match not found" });

      // Store results
      if (metricsA) {
        const profileA = await storage.getClashAgentProfile(match.agentAProfileId);
        await storage.createClashResult({ clashMatchId: matchId, agentProfileId: match.agentAProfileId, providerId: profileA?.providerId || null, ...metricsA });
      }
      if (metricsB) {
        const profileB = await storage.getClashAgentProfile(match.agentBProfileId);
        await storage.createClashResult({ clashMatchId: matchId, agentProfileId: match.agentBProfileId, providerId: profileB?.providerId || null, ...metricsB });
      }

      // Determine winner
      let winnerId: number | null = null;
      if (!matchError && metricsA?.responseLatencyMedian && metricsB?.responseLatencyMedian) {
        const ratio = metricsA.responseLatencyMedian / metricsB.responseLatencyMedian;
        if (ratio < 0.9) winnerId = match.agentAProfileId;
        else if (ratio > 1.1) winnerId = match.agentBProfileId;
      }

        completedAt: new Date(),
        recordingUrl: recordingUrl || null,

        error: matchError || null,
        winnerId,

      await storage.updateClashRunner(runner.id, { state: "idle", currentMatchId: null });

      const event = await storage.getClashEvent(match.eventId);

      }

      res.json({ success: true });

      res.status(500).json({ error: "Failed to complete" });

  // --- Event Match Scheduler (assigns next pending match to idle runner) ---
  setInterval(async () => {
    try {
      // Check for live events with pending matches
      const liveEvents = await storage.getClashEventsByStatus("live");
      for (const event of liveEvents) {
        const matches = await storage.getClashMatchesByEvent(event.id);
        const runningMatch = matches.find(m => m.status === "live" || m.status === "starting");
        if (runningMatch) continue; // Wait for current match to finish

        const nextMatch = matches.find(m => m.status === "pending");
        if (!nextMatch) continue;

        // Find idle runner in region
        const runner = await storage.getIdleClashRunner(event.region);
        if (!runner) continue;

        // Assign match to runner
        await storage.updateClashRunner(runner.id, { state: "assigned", currentMatchId: nextMatch.id });
      }

      // Health check: mark stale runners as draining
      await storage.markStaleRunnersDraining();
    } catch (err) {
      console.error("[ClashScheduler] Error:", err);
    }
  }, 10_000); // Check every 10 seconds

  // --- Schedule Cron (creates events from schedules) ---
  setInterval(async () => {
    try {
      const dueSchedules = await storage.getDueClashSchedules();
      for (const schedule of dueSchedules) {
        try {
          const matchups = schedule.matchups as { agentAProfileId: number; agentBProfileId: number; topic?: string }[];
          // Create event
          const event = await storage.createClashEvent({
            name: schedule.eventName,
            createdBy: schedule.createdBy,
            region: schedule.region,
            status: "upcoming",
            visibility: "public",
            scheduledAt: null, // Start immediately
            agoraChannelName: isAgoraConfigured() ? generateEventChannelName(0) : null,
            moderatorAgentId: null,
          });
          if (isAgoraConfigured()) {
            await storage.updateClashEvent(event.id, { agoraChannelName: generateEventChannelName(event.id) });
          }
          // Create matches
          for (let i = 0; i < matchups.length; i++) {
            await storage.createClashMatch({
              eventId: event.id,
              matchOrder: i + 1,
              agentAProfileId: matchups[i].agentAProfileId,
              agentBProfileId: matchups[i].agentBProfileId,
              topic: matchups[i].topic || "Freestyle debate",
              maxDurationSeconds: schedule.maxDurationSeconds,
              status: "pending",
              config: {},
            });
          }
          // Start event immediately
          await storage.updateClashEvent(event.id, { status: "live", startedAt: new Date() });

          // Update schedule
          if (schedule.cronExpression) {
